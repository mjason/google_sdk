def initialize(redirect_uri)
  @base_uri = "https://accounts.google.com"
  @redirect_uri = redirect_uri
  @access_type = 'offline'
  @approval_prompt = 'force'

  @client_id = GlassSdk.client_id
  @client_secret = GlassSdk.client_secret
  @scope = GlassSdk.scope
end

def login_url
  params = []
  {
    response_type: 'code',
    client_id: @client_id,
    redirect_uri: @redirect_uri,
    scope: @scope.join("+"),
    access_type: @access_type,
    approval_prompt: @approval_prompt
  }.each do |key, value|
    params << "#{key}=#{value}"
  end
  return "#{@base_uri}/o/oauth2/auth?#{params.join("&")}"      
end

def user_token(code)
  RestClient.post "#{@base_uri}/o/oauth2/token", {
    code: code,
    client_id: @client_id,
    client_secret: @client_secret,
    redirect_uri: @redirect_uri,
    grant_type: 'authorization_code'
  }
end
